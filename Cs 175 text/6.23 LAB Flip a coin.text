//6.23 LAB: Flip a coin

#include <iostream>
#include <stdlib.h>

using namespace std;

string HeadsOrTails()
{
   if(rand() % 2 == 0)
       return "heads";
   else
       return "tails";

}

int main()
{
   srand(2);
   int j;
   cin >> j;
  
   for(int i = 0; i < j; i++)
   {
       cout << HeadsOrTails() << endl;
   }
  
}
